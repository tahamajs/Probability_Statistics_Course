import  numpy as np
def sample_binomial_distribution(m, n, p):
    samples = np.random.choice([0, 1], size=(m, n), p=[1-p, p])
    sums = np.sum(samples, axis=1)
    return sums

in this function we have m number of samples that has binoial destribution with n number of trials and p probability of success becuse we have n number of bernouli destribution and we want to sum them up we can use binomial destribution for this problem out put of this function is m number of samples that has binomial destribution with n number of trials and p probability of success we have n number of bernouli destribution and we want to sum them up we can use binomial destribution for this problem .

import matplotlib.pyplot as plt

sample_means = np.zeros(len(p_values))
sample_variances = np.zeros(len(p_values))

for i, p in enumerate(p_values):
    # for all of 100 p values we generate 5000 samples of binomial destribution and get that samples mean and variance and put them in sample_means and sample_variances arrays
    samples = sample_binomial_distribution(m, n, p)
    sample_means[i] = np.mean(samples)
    sample_variances[i] = np.var(samples)

# plot the results
plt.figure(figsize=(20, 20))
plt.plot(p_values, theoretical_means, label="theoretical mean")
plt.plot(p_values, sample_means, label="practical mean")
plt.plot(p_values, theoretical_variances, label="theoretical variance ")
plt.plot(p_values, sample_variances, label="practical variance ")
plt.xlabel("P values")
plt.ylabel("samples")
plt.legend()
plt.title("theoretical and practical  values for binomial distribution")
plt.grid(True)
